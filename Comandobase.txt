Adicione console.log nos possiveis erros e avisos, altere o "COMANDO" e "NOME-DO-COMANDO" 
 para o nome do .js usado em letra maiuscula, "${interaction.user.tag}" para o usuario que interagiu e "USUARIO-MARCADO" para quem foi marcado

const { SlashCommandBuilder } = require('@discordjs/builders');
const { MessageEmbed } = require('discord.js');
const config = require('../../config.json');

console.log('[ NOME-DO-COMANDO > AVISO ] O ${interaction.user.tag}  tentou executar o comando sem a permissão necessaria.');
console.log('[ NOME-DO-COMANDO ] ${interaction.user.tag} executou o comando /comando .');

console.log('[ NOME-DO-COMANDO > AVISO ] O ${interaction.user.tag}  tentou executar o comando no USUARIO-MARCADO que possue um cargo mais alto.');
console.log('[ NOME-DO-COMANDO > ERRO ] Erro ao executar o comando');

module.exports = {
  data: new SlashCommandBuilder()
    .setName('meucomando')
    .setDescription('Descrição do meu comando'),
  async execute(interaction) {
    const embed = new MessageEmbed()
      .setTitle('Título do embed')
      .setDescription('Descrição do embed')
      .addFields(
        { name: 'Nome do campo 1', value: 'Valor do campo 1' },
        { name: 'Nome do campo 2', value: 'Valor do campo 2' },
        { name: 'Nome do campo 3', value: 'Valor do campo 3' },
      )
      .setThumbnail(config.Logo);
    await interaction.reply({ embeds: [embed] });
  },
};
